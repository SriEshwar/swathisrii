//asynchronous callback function
const setTimeout = require('timers/promises').setTimeout;


function setTimeoutAsync(delay, callback) {
  setTimeout(delay)
    .then(() => {
           callback(null, 'This message will print after 4 seconds');
    })
    .catch((err) => {
            callback(err);
    });
}

setTimeoutAsync(4000, (err, data) => {
  if (err) {
        console.error(err);
    return;
  }
    console.log(data);
});

//node events

const EventEmitter = require('events');

const eventEmitter = new EventEmitter();

const fun1 = (msg) => {
  console.log("Message from fun1: " + msg);
};

const fun2 = (msg) => {
  console.log("Message from fun2: " + msg);
};

eventEmitter.on('myEvent', fun1);
eventEmitter.prependListener('myEvent', fun2);

console.log(eventEmitter.listeners('myEvent'));
console.log(eventEmitter.listenerCount('myEvent'));

const express = require('express');
const bodyParser = require('body-parser');

const app = express();
app.use(bodyParser.urlencoded({ extended: true }));

const users = [];

app.get('/', (req, res) => {
  res.sendFile(__dirname + '/signup.html');
});

app.post('/submit_signup', (req, res) => {
  const user = {
    username: req.body.username,
    password: req.body.password,
    phone: req.body.phone,
    email: req.body.email
  };
  users.push(user);
  res.send('Signup successful!<script>alert("Signup successful!");</script>');
});

app.listen(3000, () => {
  console.log('Server is running on port 3000');
});
